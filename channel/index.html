渠道

uno支持渠道划分，用于支持不同技术体系，不同流程的业务

配置信息

按照以下说明新增lion 填写配置 申请时提供lion key即可 @daxuan.she @yongqiang.shen

{
    "code":"1", //业务不需要配置这个字段  由uno分配
    "desc": "彩虹",  //渠道名称
    "publish": 3,   //发布code   1.预发   2.下线   3.uno发布（vue1）   10.亲子发布(vue2)
    "preview":1,   //预览code  1vue1预览   2vue2预览
    "url":"/av/rainbow/${topicId}/index.html",
    "urlParams":["activityId",{"key":"param2","ifExp":"pageType=='widely-order-create'"}], //访问连接的参数 参数值必须保存在page  extendInfo 里  支持string  和obj（对象目前仅预览链接支持 ifExp控制是否带有这个参数  值取自extendInfo 
    "ci":{
      "type":"jenkins",   //jenkins or normal 
      "hostLion":"uno-web.buildUrl", //
      "configLion":"uno-web.jenkinsConfig" //jenkins ci 创建发布项需要的config的Lion 
    },
    "tagList": [
        {  //页面级别扩展信息
            "title": "页面类型",
            "type": "string",
            "key": "pageType",
            "optionList": [{
                "label": "通用类型",
                "key": "common"
                }, {
                "label": "通用预订 - 下单页",
                "key": "widely-order-create"
            }]
        }, 
        {
            "title": "页面打点",
            "type": "string",
            "key": "cid"
        },{
            "title":"sku id",
            "key":"productitemid",
            "type":"string",
            "ifExp":"pageType=='widely-order-create'"  //支持控制  值取自extendInfo 
    }],
    "formCbType":"post",  //表单点击提交 触发的行为  支持‘post’  'get'  'jsonp'  'redirect'(默认)
    "formCbReqUrl":"//uno.51ping.com/ajax/test",
     //  1.当formCbType不是默认值时需要 指定 请求的url   
     //2.组件配置指定isFormCbReqParam:true 的字段会作为参数  
     //3.请求返回的数据会保存在页面extendInfo字段里  
    "formBindUrl":"//apie.51ping.com/marketing/microwelfare/createwelfare",
    //1.表单页绑定topic 和商户 的接口 url 默认是丽人微海报的接口
    //2.组件配置指定isFormCbBindParam:true 的字段会作为参数  
    "formBindType":"post",   //表单页绑定topic 和商户 的接口  类型  默认jsonp   支持 ‘post’  'get'  'jsonp'
    "formTitle":"创建福利卡",  //表单页title  默认 编辑微海报
    "mainColor":"#FF8099",//表单页主色调 目前仅控制了提交按钮
    "formCbRedirectUrl":"//uno.51ping.com"//表单页绑定接口成功后 重定向的地址  ‘默认是预览页面’
}
