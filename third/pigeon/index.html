pegion接口

maven依赖

<dependency>
 <groupId>com.dianping.gmkt</groupId>
 <artifactId>gmkt-uno-api</artifactId>
 <version>1.0-SNAPSHOT</version>
</dependency>

bean配置

<!--UNO通用服务-->
<bean id="unoCommonService" class="com.dianping.pigeon.remoting.invoker.config.spring.ReferenceBean" init-method="init">
 <property name="url" value="http://service.dianping.com/gmktUnoService/unoCommonService_1.0.0" />
 <property name="interfaceName" value="com.dianping.gmkt.uno.api.service.GmktUnoCommonService" />
 <property name="timeout" value="2000" />
</bean>

<!--UNO模板服务-->
<bean id="unoTemplateService" class="com.dianping.pigeon.remoting.invoker.config.spring.ReferenceBean" init-method="init">
 <property name="url" value="http://service.dianping.com/gmktUnoService/unoTemplateService_1.0.0" />
 <property name="interfaceName" value="com.dianping.gmkt.uno.api.service.GmktUnoTemplateService" />
 <property name="timeout" value="2000" />
</bean>

接口

GmktUnoCommonService

/**
 * 根据topicId获取uno专题详细信息
 * @param topicId
 * @return
 */
GmktUnoResponse<UnoPage> getPage(Integer topicId);

/**
 * 根据topicIds 获取uno专题详细信息
 * 最大查询数量 60
 * @param topicId
 * @return
 */
GmktUnoResponse<Map<Integer, UnoPage>> getPages(List<Integer> topicId);

/**
 * 根据topicId删除uno专题页
 * @param topicId
 * @return
 */
GmktUnoResponse<Integer> deletePage(Integer topicId);

/**
 * 根据topicId获取cid
 * @param topicId
 * @return
 */
GmktUnoResponse<String> getCid(Integer topicId);

GmktUnoTemplateService

/**
 * 根据模板ids 获取uno专题模板详细信息
 * 最大查询数量 lion:{gmkt-uno-service.unoTemplates.querySizeLimit} 60
 * @param ids
 * @return
 */
GmktUnoResponse<Map<Integer, UnoTemplate>> getTemplates(List<Integer> ids);

/**
 * 查询符合条件的uno专题模板列表
 * 根据 businessKey 和 statusList查询
 * @param request
 * @return
 */
GmktUnoResponse<UnoCMPagination<List<UnoTemplate>>> findTemplateList(UnoTemplateListReq request);
