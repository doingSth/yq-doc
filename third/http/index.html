创建页面接口

url: /ajax/savePage  
methode: POST 
content-type: 'application/x-www-form-urlencoded'
body:
{
"topicId":123,//有则为修改  不传为新增
"name": "一分钱抽奖活动创建测试",  //必须  限制50字符
"title": "一分钱抽奖活动创建测试", //  限制50字符
"endTime":"2018-09-03",//必须 过期时间
"author": "yongqiang.shen", // 页面创建人  标示  不仅可以mis号
"bu":0,//{1: '丽人',2: '家装',4: '教育',8: '亲子',16: 'KTV&休娱',32: '结婚',0:'超管（其他）'} 若需要在uno列表页看到需传查看人对应的bu
"platform": 1,//平台1点评 2美团  必须  若无平台概念  传1
"source":1,//1: '彩虹',2: '亲子商户营销',3: '人品团',4: '微海报'  必须
"share":"{\"title\":\"全城好券\",\"image\":\"//p0.meituan.net/dprainbow/7adafc19037b0bf12fdc408f17e465d16877.png\",\"useH5Bar\":false,\"h5BarColor\":\"#ddd\"}",//分享信息 非必须
"config":"",//组件配置  非必须
"templateId":1,//模版Id  非必须
"extendInfo":"{}",//扩展信息 json 字符串   非必须
"versionConfig":"{}"//组件版本信息  非必须
}
