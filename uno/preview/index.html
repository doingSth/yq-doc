客户端监听postmsg 实时变更视图

   window.addEventListener('message',function(value){
            logger.info('预览接收到的数据',value.data);
            switch(value.data.type){
                case 'select':
                    selectComponent(value.data.index);
                    break;
                case 'config':
                    $(window).trigger('updateData', [value.data.config])

                    $allComponents.each(function(i, component){
                        renderEditable(component, value.data.config[i]);

                    });
                    clearInterval(_interval);
                    _interval = setInterval(function(){
                        $allComponents.each(function(i, component){
                            renderEmpty(component);
                        })
                    },500)


                    break;
            }

        })
